This chapter illustrates the style of these notes: we take a statement the

\section{The operator \texorpdfstring{$\dbar$}{d-bar}}

\begin{definition}\label{def:dbar}
Let $U \subseteq \C^n$ be open and $f \in C^\infty(U)$. The \emph{Dolbeault
operator} is
\[
  \dbar f = \sum_{j=1}^{n} \frac{\partial f}{\partial \bar{z}_j}\, d\bar{z}_j ,
\]
extended to $(p,q)$-forms by acting on coefficient functions and leaving the
generators $dz_I \wedge d\bar{z}_J$ fixed.
\end{definition}

\begin{remark}
Kodaira states $\dbar^2 = 0$ without proof. We supply it for functions; the
general case follows by linearity and the Leibniz rule.
\end{remark}

\begin{proposition}\label{prop:dbar-squared}
For every $f \in C^\infty(U)$ we have $\dbar(\dbar f) = 0$.
\end{proposition}

\begin{proof}
By Definition~\ref{def:dbar},
\[
  \dbar f = \sum_{j} \frac{\partial f}{\partial \bar z_j}\, d\bar z_j .
\]
Applying $\dbar$ again, differentiating each coefficient,
\[
  \dbar(\dbar f)
   = \sum_{k}\sum_{j}
     \frac{\partial^2 f}{\partial \bar z_k \, \partial \bar z_j}\,
     d\bar z_k \wedge d\bar z_j .
\]
The mixed second derivatives are symmetric in $j,k$ (Schwarz's theorem, valid
since $f$ is smooth), while $d\bar z_k \wedge d\bar z_j = -\, d\bar z_j \wedge
d\bar z_k$ is antisymmetric. Pairing the $(j,k)$ and $(k,j)$ terms,
\[
  \frac{\partial^2 f}{\partial \bar z_k \partial \bar z_j}\,
    d\bar z_k \wedge d\bar z_j
  + \frac{\partial^2 f}{\partial \bar z_j \partial \bar z_k}\,
    d\bar z_j \wedge d\bar z_k = 0 ,
\]
so every term cancels and $\dbar(\dbar f) = 0$.
\end{proof}
